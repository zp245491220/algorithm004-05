    public int climbStairs(int n) {
        int  result = 0;
        result = helper(n, result);
        return result;
    }

    public int helper(int n, int result) {
        if (n < 1) return 0;
        if (n == 1) return 1;
        if (n == 2) return 2;
        result = helper(n - 1, result) + helper(n - 2, result);
        return result;
    }

    public int climbStairs(int n) {
        int  result = 0;
        if (n <= 2) return n;
        int firstNum = 1;
        int secondNum = 2;
        for (int i = 3; i < n; i++) {
            result = firstNum + secondNum;
            firstNum = secondNum;
            secondNum = result;
        }
        return result;
    }
